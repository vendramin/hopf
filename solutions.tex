\begin{sol}{xca:coalgebra}
  For the coassociativity, 
  \begin{align*}
    \sum x_{1,1}\otimes x_{1,2}\otimes x_2 &= 
    (\Delta\otimes\id)\Delta(x)
    =(\id\otimes\Delta)\Delta(x)
    =\sum x_1\otimes x_{2,1}\otimes x_{2,2}.
  \end{align*}
  For the counit, 
  \[
    \sum\epsilon(x_1)x_2=(\epsilon\otimes\id)\Delta(x)
    =x=(\id\otimes\epsilon)\Delta(x)=\sum x_1\epsilon(x_2).
  \]
\end{sol}

\begin{sol}{xca:primitive}
  Let $x\in C$. Applying $\epsilon\otimes\id$ 
  to $\Delta(x)=x\otimes1+1\otimes x$, we get
  \begin{align*} 
    x=(\epsilon\otimes\id)\Delta(x)=\epsilon(x)1+\epsilon(1)x
    =\epsilon(x)+x.
  \end{align*}
  Thus $\epsilon(x)=0$. 
\end{sol}

\begin{sol}{xca:1.6.4a}
  Let $m\in M$. Then 
  \[
    \epsilon\cdot m=\sum\epsilon(m_1)m_0=m.
  \]
  Now let $f,g\in C^*$.  Since 
  \begin{align*}
    g\cdot (f\cdot m) &= g\cdot\left( \sum f(m_1)m_0 \right)\\
                      &= \sum g( (f(m_1)m_0)_1 )(f(m_1)m_0)_0\\
                      &= \sum f(m_1)g(m_{0,1})m_{0,0}\\
                      &= \sum f(m_1)g(m_0)m_{-1}.
  \shortintertext{and}
   (gf)\cdot m &= \sum (gf)(m_1)m_0\\
               &= \sum g(m_{1,1})f(m_{1,2})m_0\\
               &= \sum g(m_0)f(m_1)m_{-1},
  \end{align*}
  the claim follows. 
\end{sol}
